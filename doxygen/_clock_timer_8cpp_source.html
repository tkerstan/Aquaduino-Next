94 int8_t ClockTimer::check()
95 {
96  uint8_t on = 0;
97  unsigned long nowInSecs;
98  unsigned long clockTimerOnInSecs;
99  unsigned long clockTimerOffInSecs;
100 
101  switch (weekday())
102  {
103  case dowMonday:
104  if (!isMondayEnabled())
105  return 0;
106  break;
107  case dowTuesday:
108  if (!isTuesdayEnabled())
109  return 0;
110  break;
111  case dowWednesday:
112  if (!isWednesdayEnabled())
113  return 0;
114  break;
115  case dowThursday:
116  if (!isThursdayEnabled())
117  return 0;
118  break;
119  case dowFriday:
120  if (!isFridayEnabled())
121  return 0;
122  break;
123  case dowSaturday:
124  if (!isSaturdayEnabled())
125  return 0;
126  break;
127  case dowSunday:
128  if (!isSundayEnabled())
129  return 0;
130  break;
131  default:
132  return 0;
133  break;
134  }
135 
136  nowInSecs = hour() * SECS_PER_HOUR + minute() * SECS_PER_MIN + second();
137 
138  for (int i = 0; i < max_timers; i++)
139  {
140  clockTimerOnInSecs = this->m_HOn[i] * SECS_PER_HOUR
141  + this->m_MOn[i] * SECS_PER_MIN;
142  clockTimerOffInSecs = this->m_HOff[i] * SECS_PER_HOUR
143  + this->m_MOff[i] * SECS_PER_MIN;
144 
145  // Check whether entry is not hh:mm - hh:mm
146  if (clockTimerOnInSecs - clockTimerOffInSecs != 0)
147  {
148  // e.g. 08:00 - 15:00
149  if (clockTimerOnInSecs < clockTimerOffInSecs)
150  {
151  on |= (nowInSecs >= clockTimerOnInSecs && nowInSecs
152  < clockTimerOffInSecs) ? 1 : 0;
153  // e.g. 21:00 - 06:00
154  }
155  else
156  {
157  on |= (nowInSecs >= clockTimerOnInSecs || nowInSecs
158  < clockTimerOffInSecs) ? 1 : 0;
159  }
160 
161 
162  }
163  }
164 
165  return on;
166 
167 }
