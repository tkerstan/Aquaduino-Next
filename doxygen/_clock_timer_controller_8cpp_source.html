57 uint16_t ClockTimerController::serialize(void* buffer, uint16_t size)
58 {
59  uint8_t i;
60  uint8_t clockTimers = MAX_CLOCKTIMERS;
61  uint8_t* charBuffer = (uint8_t*) buffer;
62  uint16_t pos = 0;
63  uint16_t timerSize = 0;
64  int32_t left = size;
65 
66  charBuffer[0] = clockTimers;
67 
68  if (sizeof(m_ActuatorMapping) + sizeof(uint8_t) > size)
69  return 0;
70 
71  pos += sizeof(uint8_t);
72  memcpy(charBuffer + pos, m_ActuatorMapping, sizeof(m_ActuatorMapping));
73  pos += sizeof(m_ActuatorMapping);
74  for (i = 0; i < MAX_CLOCKTIMERS; i++)
75  {
76  left = ((int32_t) size) - pos - 2;
77  if (left < 0)
78  return 0;
79  timerSize = m_Timers[i].serialize(charBuffer + pos + 2, left);
80  charBuffer[pos] = (timerSize & 0xFF00) >> 8;
81  charBuffer[pos + 1] = timerSize & 0xFF;
82  pos += timerSize + 2;
83  }
84  return pos;
85 }

87 uint16_t ClockTimerController::deserialize(void* data, uint16_t size)
88 {
89  uint8_t i;
90  uint8_t clockTimers = 0;
91  uint8_t* charBuffer = (uint8_t*) data;
92  uint16_t pos = 0;
93  uint16_t timerSize = 0;
94 
95  clockTimers = charBuffer[0];
96 
97  if (clockTimers != MAX_CLOCKTIMERS)
98  return 0;
99 
100  pos += sizeof(uint8_t);
101 
102  memcpy(m_ActuatorMapping, charBuffer + pos, sizeof(m_ActuatorMapping));
103 
104  pos += sizeof(m_ActuatorMapping);
105  for (i = 0; i < MAX_CLOCKTIMERS; i++)
106  {
107  timerSize = charBuffer[pos] << 8;
108  timerSize += charBuffer[pos + 1];
109  pos += 2;
110 
111  if (m_Timers[i].deserialize(charBuffer + pos, timerSize) == 0)
112  return 0;
113  pos += timerSize;
114  }
115  return pos;
116 }

434 int8_t ClockTimerController::processPost(WebServer* server,
435  WebServer::ConnectionType type,
436  char* url)
437 {
438  int8_t repeat;
439  int8_t x, y;
440  int8_t input;
441  int16_t val;
442  char name[16], value[16];
443 
444  //No SubURL given. The post target is the clocktimer form and
445  //not the timer select form. Thus the selected DOW are cleared before and
446  //the selected DOW will be extracted in the do while loop.
447  if (url == NULL)
448  m_Timers[m_SelectedTimer].disableAllDays();
449 
450  do
451  {
452  repeat = server->readPOSTparam(name, 16, value, 16);
453  if (url != NULL && strcmp_P(url, pgm_url_select) == 0)
454  {
455  if (strcmp_P(name,
456  (PGM_P) pgm_read_word(&(template_clocktimercontroller_inputs[CLOCKTIMERCONTROLLER_I_TIMER])))
457  == 0)
458  {
459  m_SelectedTimer = atoi(value);
460  }
461  }
462  {
463 
464  if (strcmp_P(name,
465  (PGM_P) pgm_read_word(&(template_clocktimercontroller_inputs[CLOCKTIMERCONTROLLER_I_ACTUATOR])))
466  == 0)
467  {
468  val = atoi(value);
469  if (val != -1)
470  {
471  //if (!isMapped(atoi(value)))
472  m_ActuatorMapping[m_SelectedTimer] = atoi(value);
473  }
474  else
475  {
476  m_ActuatorMapping[m_SelectedTimer] = -1;
477  }
478  }
479  else if (strcmp_P(name,
480  (PGM_P) pgm_read_word(&(template_clocktimercontroller_inputs[CLOCKTIMERCONTROLLER_I_DOW])))
481  == 0)
482  {
483  val = atoi(value);
484  switch (val)
485  {
486  case 0:
487  m_Timers[m_SelectedTimer].enableMonday();
488  break;
489  case 1:
490  m_Timers[m_SelectedTimer].enableTuesday();
491  break;
492  case 2:
493  m_Timers[m_SelectedTimer].enableWednesday();
494  break;
495  case 3:
496  m_Timers[m_SelectedTimer].enableThursday();
497  break;
498  case 4:
499  m_Timers[m_SelectedTimer].enableFriday();
500  break;
501  case 5:
502  m_Timers[m_SelectedTimer].enableSaturday();
503  break;
504  case 6:
505  m_Timers[m_SelectedTimer].enableSunday();
506  break;
507 
508  }
509  }
510  else
511  {
512  input = atoi(name);
513  x = (input - 1) / 4;
514  y = input % 4;
515 
516  //Ignore input with name 0!
517  if (input != 0)
518  {
519  switch (y)
520  {
521  case 0:
522  m_Timers[m_SelectedTimer].setMinuteOff(x,
523  atoi(value));
524 
525  break;
526  case 1:
527  m_Timers[m_SelectedTimer].setHourOn(x, atoi(value));
528 
529  break;
530  case 2:
531  m_Timers[m_SelectedTimer].setMinuteOn(x,
532  atoi(value));
533 
534  break;
535  case 3:
536  m_Timers[m_SelectedTimer].setHourOff(x,
537  atoi(value));
538  break;
539  }
540  }
541  }
542  }
543 
544  } while (repeat);
545 
546  return 0;
547 }
