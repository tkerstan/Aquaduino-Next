121 int8_t LevelController::run()
122 {
123  static unsigned long lastTime = 0;
124  unsigned long millisNow = 0;
125  long delay_low_millis = 0;
126  long delay_high_millis = 0;
127  long timeout_millis = 0;
128  long deltaTSwitch = 0;
129  int8_t sensor_val = 0;
130 
131  if (m_Sensor < 0 || m_Sensor >= MAX_SENSORS)
132  return -1;
133 
134  sensor_val = __aquaduino->getSensorValue(m_Sensor);
135  millisNow = millis();
136  deltaTSwitch = millisNow - lastTime;
137 
138  delay_high_millis = 1000 * (long) m_Delayh;
139  delay_low_millis = 1000 * (long) m_Delayl;
140  timeout_millis = 1000 * (long) m_Timeout;
141 
142  //Check for overflow while processing refill
143  if (m_State != LEVELCONTROLLER_STATE_OK && lastTime > millisNow)
144  deltaTSwitch = LONG_MAX - lastTime + millisNow;
145 
146  switch (m_State)
147  {
148  case LEVELCONTROLLER_STATE_OK:
149  allMyActuators((int8_t) 0);
150  if (sensor_val == HIGH)
151  {
152  m_State = LEVELCONTROLLER_STATE_DEBOUNCE;
153  lastTime = millisNow;
154  }
155  break;
156  case LEVELCONTROLLER_STATE_DEBOUNCE:
157  if (sensor_val == HIGH && deltaTSwitch > delay_high_millis)
158  {
159  allMyActuators((int8_t) 1);
160  lastTime = millisNow;
161  m_State = LEVELCONTROLLER_STATE_REFILL;
162  }
163  else if (sensor_val == LOW)
164  {
165  m_State = LEVELCONTROLLER_STATE_OK;
166  }
167  break;
168  case LEVELCONTROLLER_STATE_REFILL:
169  if (sensor_val == LOW)
170  {
171  m_State = LEVELCONTROLLER_STATE_OVERRUN;
172  lastTime = millisNow;
173  }
174  else if (sensor_val == HIGH && deltaTSwitch > timeout_millis)
175  {
176  m_State = LEVELCONTROLLER_STATE_REFILL_TIMEOUT;
177  allMyActuators((int8_t) 0);
178  }
179  break;
180  case LEVELCONTROLLER_STATE_OVERRUN:
181  if (sensor_val == LOW && deltaTSwitch > delay_low_millis)
182  {
183  m_State = LEVELCONTROLLER_STATE_OK;
184  allMyActuators((int8_t) 0);
185  }
186  else if (sensor_val == HIGH)
187  {
188  m_State = LEVELCONTROLLER_STATE_REFILL;
189  }
190  break;
191  case LEVELCONTROLLER_STATE_REFILL_TIMEOUT:
192  return 0;
193  default:
194  return 0;
195  }
196  return 1;
197 }
