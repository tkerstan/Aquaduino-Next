39 uint8_t EthernetUDP::begin(uint16_t port) {
40  if (_sock != MAX_SOCK_NUM)
41  return 0;
42 
43  for (int i = 0; i < MAX_SOCK_NUM; i++) {
44  uint8_t s = W5100.readSnSR(i);
45  if (s == SnSR::CLOSED || s == SnSR::FIN_WAIT) {
46  _sock = i;

50

52  return 0;
53 
54  _port = port;

118 int EthernetUDP::parsePacket()
119 {
120  // discard any remaining bytes in the last packet
121  flush();
122 
123  if (W5100.getRXReceivedSize(_sock) > 0)
124  {
125  //HACK - hand-parse the UDP packet using TCP recv method
126  uint8_t tmpBuf[8];
127  int ret =0;
128  //read 8 header bytes and get IP and port from it
129  ret = recv(_sock,tmpBuf,8);
130  if (ret > 0)
131  {
132  _remoteIP = tmpBuf;
133  _remotePort = tmpBuf[4];
134  _remotePort = (_remotePort << 8) + tmpBuf[5];
135  _remaining = tmpBuf[6];
136  _remaining = (_remaining << 8) + tmpBuf[7];
137 
138  // When we get here, any remaining bytes are the data
139  ret = _remaining;
140  }
141  return ret;
142  }
143  // There aren't any packets available
144  return 0;
145 }
