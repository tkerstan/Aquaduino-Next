76 uint16_t DigitalOutput::deserialize(void* data, uint16_t size)
77 {
78  uint8_t* bPtr = (uint8_t*) data;
79  uint16_t mySize = sizeof(m_OnValue) + sizeof(m_OffValue);
80 
81  if (size < mySize)
82  return 0;
83 
84  memcpy(&m_OnValue, bPtr, sizeof(m_OnValue));
85  bPtr += sizeof(m_OnValue);
86  memcpy(&m_OffValue, bPtr, sizeof(m_OffValue));
87  bPtr += sizeof(m_OffValue);
88  memcpy(&m_Pin, bPtr, sizeof(m_Pin));
89  bPtr += sizeof(m_Pin);
90  memcpy(&m_On, bPtr, sizeof(m_On));
91  bPtr += sizeof(m_On);
92  memcpy(&m_DutyCycle, bPtr, sizeof(m_DutyCycle));
93  pinMode(m_Pin, OUTPUT);
94 
95  if (supportsPWM())
96  setPWM(m_DutyCycle);
97  else if (m_On)
98  forceOn();
99  else
100  forceOff();
101 
102  return mySize;

199 void DigitalOutput::setPWM(float dutyCycle)
200 {
201  if (supportsPWM() && dutyCycle <= 1.0)
202  {
203  m_DutyCycle = dutyCycle;
204  if (m_OnValue == 0)
205  analogWrite(m_Pin, (uint8_t) ((1.0 - dutyCycle) * 255));
206  else
207  analogWrite(m_Pin, (uint8_t) (dutyCycle * 255));
208  if (m_DutyCycle > 0)
209  m_On = 1;
210  else
211  m_On = 0;
212  }
213 }

235 #ifdef FEATURE_WEBIF
236 int8_t DigitalOutput::showWebinterface(WebServer* server,
237  WebServer::ConnectionType type,
238  char* url)
239 {
240  File templateFile;
241  TemplateParser* parser;
242  int16_t matchIdx;
243  char templateFileName[template_digitaloutput_fnsize];
244  strcpy_P(templateFileName, template_digitaloutput_fname);
245  float dC = -1.0;
246 
247  if (type == WebServer::POST)
248  {
249  int8_t repeat;
250  char name[16], value[16];
251  do
252  {
253  repeat = server->readPOSTparam(name, 16, value, 16);
254  if (strcmp_P(name,
255  (PGM_P) pgm_read_word(&(template_digitaloutput_inputs[DO_I_ON])))
256  == 0)
257  {
258  m_OnValue = atoi(value);
259  m_OffValue = 1 - m_OnValue;
260  }
261  else if (strcmp_P(name,
262  (PGM_P) pgm_read_word(&(template_digitaloutput_inputs[DO_I_PIN])))
263  == 0)
264  {
265  m_Pin = atoi(value);
266  pinMode(m_Pin, OUTPUT);
267  }
268  else if (strcmp_P(name,
269  (PGM_P) pgm_read_word(&(template_digitaloutput_inputs[DO_I_PWM])))
270  == 0)
271  {
272  dC = atof(value);
273  if (dC < 0.0)
274  dC = 0;
275  else if (dC > 100.0)
276  dC = 100.0;
277  dC /= 100;
278  }
279 
280  } while (repeat);
281  if (dC >= 0)
282  setPWM(dC);
283  server->httpSeeOther(this->m_URL);
284  }
285  else
286  {
287  server->httpSuccess();
288  parser = __aquaduino->getTemplateParser();
289  templateFile = SD.open(templateFileName, FILE_READ);
290  while ((matchIdx =
291  parser->processTemplateUntilNextMatch(&templateFile,
292  template_digitaloutput,
293  template_digitaloutput_elements,
294  server))
295  >= 0)
296  {
297  switch (matchIdx)
298  {
299  case DO_ANAME:
300  server->print(getName());
301  break;
302  case DO_PIN_NAME:
303  server->print((const __FlashStringHelper *) (pgm_input_pin));
304  break;
305  case DO_PIN_VAL:
306  server->print(m_Pin);
307  break;
308  case DO_PIN_SIZE:
309  server->print(3);
310  break;
311  case DO_PIN_MAXLENGTH:
312  server->print(2);
313  break;
314  case DO_PWM_NAME:
315  server->print((const __FlashStringHelper *) (pgm_input_pwm));
316  break;
317  case DO_PWM_VAL:
318  if (supportsPWM())
319  {
320  server->print(getPWM() * 100);
321  }
322  else
323  {
324  server->print(F("No PWM"));
325  }
326  break;
327  case DO_PWM_SIZE:
328  server->print(7);
329  break;
330  case DO_PWM_MAXLENGTH:
331  server->print(6);
332  break;
333  case DO_ON_NAME:
334  server->print((const __FlashStringHelper *) (pgm_input_on));
335  break;
336  case DO_ON_OPTIONS:
337  parser->selectListOption("LOW", "0", m_OnValue == 0, server);
338  parser->selectListOption("HIGH", "1", m_OnValue == 1, server);
339  break;
340  }
341  }
342 
343  templateFile.close();
344  }
345  return true;

347 }
