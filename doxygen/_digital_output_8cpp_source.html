65 uint16_t DigitalOutput::deserialize(Stream* s)
66 {
67  uint16_t mySize = sizeof(m_OnValue) + sizeof(m_OffValue);
68 
69  if (s->available() < mySize)
70  return 0;
71 
72  s->readBytes((char*)&m_OnValue, sizeof(m_OnValue));
73  s->readBytes((char*)&m_OffValue, sizeof(m_OffValue));
74  s->readBytes((char*)&m_Pin, sizeof(m_Pin));
75  s->readBytes((char*)&m_On, sizeof(m_On));
76  s->readBytes((char*)&m_DutyCycle, sizeof(m_DutyCycle));
77 
78  if (supportsPWM())
79  setPWM(m_DutyCycle);
80  else if (m_On)
81  forceOn();
82  else
83  forceOff();
84 
85  return mySize;
86 }

182 void DigitalOutput::setPWM(float dutyCycle)
183 {
184  if (supportsPWM() && dutyCycle <= 1.0)
185  {
186  m_DutyCycle = dutyCycle;
187  if (m_OnValue == 0)
188  analogWrite(m_Pin, (uint8_t) ((1.0 - dutyCycle) * 255));
189  else
190  analogWrite(m_Pin, (uint8_t) (dutyCycle * 255));
191  if (m_DutyCycle > 0)
192  m_On = 1;
193  else
194  m_On = 0;
195  }
196 }

253  == 0)
254  {
255  dC = atof(value);
256  if (dC < 0.0)
257  dC = 0;
258  else if (dC > 100.0)
259  dC = 100.0;
260  dC /= 100;
261  }
262 
263  } while (repeat);
264  if (dC >= 0)
265  setPWM(dC);
266  server->httpSeeOther(this->m_URL);

279  {

283  server->print(getName());
284  break;
285  case DO_PIN_NAME:
286  server->print((const __FlashStringHelper *) (pgm_input_pin));

291  case DO_PIN_SIZE:

293  break;
294  case DO_PIN_MAXLENGTH:
