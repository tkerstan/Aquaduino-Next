103 uint16_t DS18S20::convertToRaw(uint8_t* data, uint8_t size, int8_t type)
104 {
105  unsigned int raw = (data[1] << 8) | data[0];
106  if (type == 1)
107  {
108  raw = raw << 3; // 9 bit resolution default
109  if (data[7] == 0x10)
110  {
111  // count remain gives full 12 bit resolution
112  raw = (raw & 0xFFF0) + 12 - data[6];
113  }
114  }
115  else
116  {
117  byte cfg = (data[4] & 0x60);
118  if (cfg == 0x00)
119  raw = raw << 3; // 9 bit resolution, 93.75 ms
120  else if (cfg == 0x20)
121  raw = raw << 2; // 10 bit res, 187.5 ms
122  else if (cfg == 0x40)
123  raw = raw << 1; // 11 bit res, 375 ms
124  // default is 12 bit resolution, 750 ms conversion time
125  }
126  return raw;
127 }
