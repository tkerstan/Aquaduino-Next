120 int8_t TemperatureController::run()
121 {
122  float temp;
123  Actuator *actuator1, *actuator2;
124 
125  if (m_Sensor == -1 || (m_Actuator1 == -1 && m_Actuator2 == -1))
126  return -1;
127 
128  temp = aquaduino->getSensorValue(m_Sensor);
129  actuator1 = aquaduino->getActuator(m_Actuator1);
130  actuator2 = aquaduino->getActuator(m_Actuator2);
131 
132  if (actuator1)
133  {
134  if (temp < m_RefTemp1)
135  {
136  actuator1->on();
137  m_Heating = 1;
138  }
139  else if (m_Heating && (temp > m_RefTemp1 + m_Hysteresis1))
140  {
141  actuator1->off();
142  m_Heating = 0;
143  }
144  }
145 
146  if (actuator2)
147  {
148  if (temp > m_RefTemp2)
149  {
150  actuator2->on();
151  m_Cooling = 1;
152  }
153  else if (actuator2 && m_Cooling && temp < m_RefTemp2 - m_Hysteresis1)
154  {
155  actuator2->off();
156  m_Cooling = 0;
157  }
158  }
159 
160  return true;
161 }
