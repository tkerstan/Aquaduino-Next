123 int8_t TemperatureController::run()
124 {
125  float temp;
126  Actuator *actuator1, *actuator2;
127 
128  if (m_Sensor == -1 || (m_Actuator1 == -1 && m_Actuator2 == -1))
129  return -1;
130 
131  temp = __aquaduino->getSensorValue(m_Sensor);
132  actuator1 = __aquaduino->getActuator(m_Actuator1);
133  actuator2 = __aquaduino->getActuator(m_Actuator2);
134 
135  if (actuator1)
136  {
137  if (temp < m_RefTemp1)
138  {
139  actuator1->on();
140  m_Heating = 1;
141  }
142  else if (m_Heating && (temp > m_RefTemp1 + m_Hysteresis1))
143  {
144  actuator1->off();
145  m_Heating = 0;
146  }
147  }
148 
149  if (actuator2)
150  {
151  if (temp > m_RefTemp2)
152  {
153  actuator2->on();
154  m_Cooling = 1;
155  }
156  else if (actuator2 && m_Cooling && temp < m_RefTemp2 - m_Hysteresis1)
157  {
158  actuator2->off();
159  m_Cooling = 0;
160  }
161  }
162 
163  return true;
164 }
