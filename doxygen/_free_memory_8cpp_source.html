41 /* Calculates the size of the free list */
42 int freeListSize() {
43  struct __freelist* current;
44  int total = 0;
45 
46  for (current = __flp; current; current = current->nx) {
47  total += 2; /* Add two bytes for the memory block's header */
48  total += (int) current->sz;
49  }
50 
51  return total;
52 }

54 int freeMemory() {

56 
57  if ((int)__brkval == 0) {
58  free_memory = ((int)&free_memory) - ((int)&__heap_start);
59  } else {

61  free_memory += freeListSize();
